   private Map<String,CpSpotRateInfo> something(ConcurrentMap<String,CpSpotRateInfo> doneRate , Map<String,CpSpotRateInfo> newRateMap){
    	
    	Map<String,CpSpotRateInfo> m = new HashMap<String, CpSpotRateInfo>();
    	
    	Set<Entry<String,CpSpotRateInfo>> s = (Set<Entry<String,CpSpotRateInfo>>)newRateMap.entrySet();
    	for (Entry<String,CpSpotRateInfo> element : s) {
    		if(doneRate.containsKey(element.getKey())){
    			if(doneRate.get(element.getKey()).getFxPriceId().equals(
    					element.getValue().getFxPriceId())){
    				newRateMap.remove(element.getKey());
    			}else{
					
					
    				m.entrySet().add(element);
//    				m_innerRateCache.putParityCheckRate(element.getValue());
    			}
    		}else{
    			doneRate.put(element.getKey(), element.getValue());
				
				
    			m.entrySet().add(element);
//    			m_innerRateCache.putParityCheckRate(element.getValue());
    		}
		}
    	return m;
    }
